二 通过Ktor实现网络请求

什么是Ktor

Ktor 是一个框架，可以轻松构建连接的应用程序——Web 应用程序、HTTP 服务、移动和浏览器应用程序。现代连接的应用程序需要异步才能为用户提供最佳体验，而 Kotlin 协程提供了非常棒的工具，可以以简单直接的方式实现。Ktor 的目标是为连接的应用程序提供端到端的多平台框架

HttpClient 是 Ktor中客户端发起HTTP请求的实例对象，可以在不同的平台上使用，包括 JVM、Android、JavaScript和Native。

特定平台可能需要处理网络请求的特定引擎。例如，您可以将Apache or Jetty用于 JVM 应用程序，OkHttp或者Android用于 Android，Curl用于面向 Kotlin/Native 的桌面应用程序，等等

不同的引擎 可能具有特定的功能并提供不同的配置选项

Ktor 提供了丰富的插件 来实现网络请求的一些拓展功能，诸如日志记录、序列化或授权之类的事情

        }
    }
}

object LoginService {
    val tag = "LoginService"

    // Ktor Client 端 发起请求的实例对象HttpClient
    val httpClient: HttpClient = httpClient {
        //日志插件
        install(Logging){
            level = LogLevel.HEADERS
            logger = object : Logger {
                override fun log(message: String) {

                }
            }
        }
        // 请求失败重试插件
        install(HttpRequestRetry) {
            maxRetries = 2
            retryIf{ request,response ->
                !response.status.isSuccess()
            }
            delayMillis { retry ->
                retry * 1000L
            }
            // Retry conditions
            modifyRequest { request ->
                request.headers.append("x-retry-count", retryCount.toString())
            }
        }
        // 序列化插件
        install(ContentNegotiation){
            // 将HttpResponse  json内容反序列化为实体对象
            json(Json {
                prettyPrint = true
                isLenient = true
                ignoreUnknownKeys = true
            })
        }
        // http response校验插件
        HttpResponseValidator {
               // http 错误统一拦截器
               handleResponseExceptionWithRequest{exception, request ->
                     val clientException = exception as? ClientRequestException ?: return@handleResponseExceptionWithRequest
                     val exceptionResponse = exception.response
                     // 404 Not Found
                     if (exceptionResponse.status == HttpStatusCode.NotFound){
                         val exceptionResponseText = exceptionResponse.bodyAsText()
                         throw MissingPageException(exceptionResponse,exceptionResponseText)
                     }
                  }
               }
        }.also {
         // 初始化Napier 日志服务
         initLogger() 
        }
}

2.3 在commonMain kotlin package下新建LoginService 实现登录业务

object LoginService {
    val tag = "LoginService"
    // 服务器ip地址
    val serverIP = "http://10.250.46.235:8089/"
    /**
     * http get request 
     */ 
    suspend fun realLogin(userName: String, userPassword: String): LoginResult {
        val validator = LoginDataValidator.verify(userName,userPassword)
        if(!validator.first){
            return LoginResult.loginFail(userName,validator.second)
        }
        val response: HttpResponse = httpClient.get("https://wireless-hipac.oss-cn-hangzhou.aliyuncs.com/android/temp/exit_users.json")
        Napier.d("response=$response", tag = tag)
        // 将HttpResponse json内容转换为 Model实体对象
        val loginModel:  BaseResponse<List<User>> = response.body()
        Napier.d("loginModel=$loginModel", tag = tag)
        val exitUsers = loginModel.data as List<User>
        var success = false
        var msg = "您还未注册,请先注册之后再登录哟❤"
        for (i in exitUsers.indices){
            if (userName.equals(exitUsers[i].userName, ignoreCase = true)){
                if (exitUsers[i].userPass == userPassword){
                    success = true
                }else{
                   msg = "密码错误❎"
                }
                break
            }
        }
        return if (success) LoginResult.loginSuccess(userName) else LoginResult.loginFail(userName,msg)
    }
}
  • HTTP POST 请求实现普通用户登录/admin登录
object LoginService {
    val tag = "LoginService"
    // 服务器ip地址
    val serverIP = "http://10.250.46.235:8089/"

    /**
     * 用户登录
     * http post 请求，Content-Type: application/x-www-form-urlencoded
     */
    suspend fun loginFromServer(userName: String, userPassword: String): LoginResult {
        val validator = LoginDataValidator.verify(userName,userPassword)
        if(!validator.first){
            return LoginResult.loginFail(userName,validator.second)
        }
        val url ="${serverIP}login"
        val params = Parameters.build {
            append("userName",userName)
            append("userPass",userPassword)
        }
        // form 表单提交
        val response: HttpResponse = httpClient.submitForm(url = url, formParameters = params ){
            method = HttpMethod.Post
        }
        Napier.d("response=$response", tag = tag)
        // HttpClient 安装 ContentNegotiation插件，配合Json 可实现HttpResponse 反序列化为实体对象
        val loginModel:  BaseResponse<User> = response.body()
        Napier.d("loginModel=$loginModel", tag = tag)
        return LoginResult.convert(loginModel)
    }
    
    /**
     * admin登录
     * http post 请求，Content-Type: application/json
     */
    suspend fun loginAdmin(userName: String, userPassword: String): BaseResponse<List<User>> {
        val validator = LoginDataValidator.verify(userName,userPassword)
        if (!validator.first){
            return BaseResponse(code = 900, success = false, message = validator.second)
        }
        val url ="${serverIP}login/admin"
        val response : HttpResponse = httpClient.post(url){
            contentType(ContentType.Application.Json)
            setBody(User(userName,userPassword,"admin",0))
        }
        Napier.d("response=$response", tag = tag)
        val model = response.body<BaseResponse<List<User>>>()
        Napier.d("model=$model", tag = tag)
        return model

    }

    suspend fun getResourceFromServer(): String {
        val response : HttpResponse = httpClient.get("${serverIP}ktor.jpg"){
            onDownload { bytesSentTotal, contentLength ->
                Napier.d("Received $bytesSentTotal bytes from $contentLength", tag = tag)
            }
        }
        val responseBody: ByteArray = response.body()
        val saveBytesToPicture = saveBytesToPicture(responseBody)
        Napier.d("saveBytesToPicture=$saveBytesToPicture", tag = tag)
        return saveBytesToPicture
    }
}

最后封装统一的数据库管理类 DatabaseManager

package cn.hipac.hello.compose.login

import cn.hipac.hello.compose.login.model.User
import cn.hipac.login.LoginDatabase

object DatabaseManager {
private val KEY = "DatabaseManager"

    private val dbCache = HashMap<String,LoginDatabase>()

    fun initDatabase(driverFactory: DriverFactory) {
        createDatabase(driverFactory)
    }

    private fun createDatabase(driverFactory: DriverFactory): LoginDatabase {
        val driver = driverFactory.createDriver()
        val db = LoginDatabase(driver)
        dbCache[KEY] = db
        return db
    }

    fun getDatabase(): LoginDatabase? {
    return dbCache[KEY]
    }
    
    /**
     * 登录用户存储
     */
    fun insertLoginUser(user: User?){
       user?.let {
          val exitUser = getDatabase()?.userQueries?.queryByName(it.userName)?.executeAsOneOrNull()
          if (exitUser == null || exitUser.userName != it.userName){
             getDatabase()?.userQueries?.insertUser(it.userName,it.userPass,it.userRole,it.age)
          }
       }
    }

}

          lifecycleScope.launch {
               withContext(Dispatchers.IO) {
                    val job = async { LoginService.loginFromServer(email, password) }
                    val loginResult = job.await()
                    withContext(Dispatchers.Main) {
                        handleLoginResult(loginResult)
                    }
               }
         }
    }


   fun handleLoginResult(result: LoginResult) {
         if (result.success){
             startActivity(Intent(this,HomeActivity::class.java))
             finish()
             val user = result.user
             DatabaseManager.insertLoginUser(user)
         }else {
             Snackbar.make(binding.btnLogin, "${result.msg}", Snackbar.LENGTH_LONG)
             .setAction("Action", null).show()
             val databaseQueries = (application as MainApp).database.userQueries
             val exitUsers = databaseQueries.queryAllUsers().executeAsList()
             exitUsers.forEach {
               Log.d("MainActivity","user=${it.toString()}")
             }
         }
   }
}

import SwiftUI
import login

struct ContentView: View {
@State private var userName: String = ""
@State private var password: String = ""

    var body: some View {
        NavigationView {
        VStack(spacing:15){
            TextField("email",text:$userName)
                .multilineTextAlignment(/*@START_MENU_TOKEN@*/.leading/*@END_MENU_TOKEN@*/)
                .frame(height:50)
                .textFieldStyle(RoundedBorderTextFieldStyle())
                .cornerRadius(5)
            SecureField("password",text:$password)
                .frame(height:50)
                .textFieldStyle(RoundedBorderTextFieldStyle())
                .cornerRadius(5)
            Button(action: {login()}, label: {
                Text("登录")
            })
            .foregroundColor(Color.white)
            .frame(width: 300, height: 40, alignment: /*@START_MENU_TOKEN@*/.center/*@END_MENU_TOKEN@*/)
            .background(Color.blue)
            .cornerRadius(5)
           
        }.padding(.all)
     }
    }
    
    func login() {
        Toast.makeText("正在登录，请稍后...").show()
        print("userName",self.userName)
        print("password",self.password)
        loginSuspend()
    }

    func loginSuspend() {
        DispatchQueue.global().async {
            print("Thread.mian",Thread.isMainThread,Thread.current)
        }
        DispatchQueue.main.asyncAfter(deadline: DispatchTime.now()+4, execute: {
        print("Thread.main",Thread.isMainThread,Thread.current)
        let loginResult: Void =  LoginService().loginFromServer(userName:self.userName,userPassword:self.password, completionHandler:{(data,error) in
            if(error == nil){
               DispatchQueue.main.async {self.handleLogin(result: data)}
            }else {
               print(error ?? "unkown error")
               DispatchQueue.main.async {Toast.makeText( "出错了").show()}
            }
         })
        print("loginResult",loginResult)
        })
    }

    func handleLogin(result: LoginResult?) {
            if let result = result{
                if (result.success) {
                    print("login success ,user=",result.user ?? "")
                    goToHome()
                    let loginDatabase = DatabaseManager.init().getDatabase()
                    print("handleLogin database=",loginDatabase ?? "")
                    DatabaseManager.init().insertLoginUser(user: result.user)
                    DispatchQueue.main.asyncAfter(deadline: DispatchTime.now()+2, execute: {
                       readUser()
                    })
                }else {
                    Toast.makeText(result.msg ?? "").show()
                }
            }
    }


    func goToHome(){
        if let window = UIApplication.shared.windows.first{
           window.rootViewController = UIHostingController(rootView: HomePageView())
           window.makeKeyAndVisible()
        }
    }

    func readUser() {
        let loginDatabase = DatabaseManager.init().getDatabase()
        if(loginDatabase != nil){
            let records = loginDatabase!.userQueries.queryAllUsers().executeAsList()
            print("record",records)
        }
    }
    
}
