二 实现登录业务逻辑

2.1 在commonMain kotlin package下新建 LoginDataValidator工具类

object LoginDataValidator {
    /*
     * 校验 登录用户输入账号密码的合法性
     */
    fun verify(userName: String?, userPassword: String?) : Pair<Boolean,String>{
        if (userName.isNullOrEmpty()){
            return Pair(false,"用户名不能为空")
        }
        if (userPassword.isNullOrEmpty()){
            return Pair(false,"密码不能为空")
        }
        return Pair(true,"校验通过")
    }
}

2.2 在commonMain kotlin package下新建登录结果数据模型 LoginResult 和 User

package cn.hipac.hello.compose.login.model

import kotlinx.serialization.Serializable

@Serializable
data class LoginResult(val success: Boolean, val user: User?, val linkUrl: String?, val msg: String?) {

    companion object {
        fun loginSuccess(userName: String?) : LoginResult {
            return LoginResult(true, User(userName ?: "unknown",null,"manager",0),"http://www.baidu.com","success")
        }

        fun loginFail(userName: String?,msg: String?): LoginResult {
            return LoginResult(false, User(userName ?: "unknown",null,"unknown",0),"http://www.baidu.com",msg ?: "用户名或者密码错误")
        }

        fun convert(model: BaseResponse<User>): LoginResult {
            return LoginResult(model.success ?: false,model.
            data, linkUrl = "http://www.baidu.com", msg = model.message ?: "unknown")
        }
    }
}

2.3 在commonMain kotlin package下新建LoginService 实现登录功能

object LoginService {
    val tag = "LoginService"
    val taskPool = HashMap<String,()-> Unit>()

    /**
     * 模拟登录，账号密码合法性校验通过，直接返回结果
     */
    fun login(userName: String?, userPassword: String?): LoginResult {
        val validator = LoginDataValidator.verify(userName, userPassword)
        return if (validator.first) {
            LoginResult.loginSuccess(userName)
        } else {
            LoginResult.loginFail(userName, validator.second)
        }
    }

    /**
     * 模拟网络请求登录，延迟3s后返回结果，该函数是suspend函数，通过异步执行
     */
    suspend fun loginAsync(userName: String?, userPassword: String?): LoginResult {
        // 模拟耗时任务
        delay(3000)
        val validator = LoginDataValidator.verify(userName, userPassword)
        return if (validator.first) {
            LoginResult.loginSuccess(userName)
        } else {
            LoginResult.loginFail(userName, validator.second)
        }
    }

    /**
     * 模拟添加任务，并执行任务返回执行结果
     */
    fun doWork(taskId: Int, task : ()-> Unit, callback: ServiceCallBack?) {
        LoginService.taskPool["$taskId"] = task
        try {
            task()
            callback?.workDone(true)
        }catch (e: Exception){
            Napier.e("doWork() error=$e", tag = tag)
            callback?.workDone(false)
            throw e
        }
    }

    interface ServiceCallBack {

        fun workDone(success: Boolean)
    }
}
  • 本节中，我们暂时没有引入网络库的概念，所以通过延时来模拟登录过程

三 实现android和iOS平台UI开发和模拟登录

3.1 实现android平台登录界面UI

class MainActivity : AppCompatActivity() {

    private lateinit var appBarConfiguration: AppBarConfiguration
    private lateinit var binding: ActivityMainBinding

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)

        binding = ActivityMainBinding.inflate(layoutInflater)
        setContentView(binding.root)

        binding.btnLogin.setOnClickListener { view ->
            Snackbar.make(view, "正在登录，请稍后...", Snackbar.LENGTH_LONG)
                .setAction("Action", null).show()
            val email = binding.etEmail.text.toString()
            val password = binding.etPassword.text.toString()
            lifecycleScope.launch{
                // IO线程发起登录 
                withContext(Dispatchers.IO){
                    val loginJob = async<LoginResult> {
                        LoginService.loginAsync(email,password)
                    }
                    val loginResult = loginJob.await()
                    withContext(Dispatchers.Main){
                        // 主线程处理登录结果
                        handleLoginResult(loginResult)
                    }
                }
            }
    }

    fun handleLoginResult(result: LoginResult) {
        if (result.success){
            startActivity(Intent(this,HomeActivity::class.java))
            finish()
        }else {
            Snackbar.make(binding.btnLogin, "${result.msg}", Snackbar.LENGTH_LONG)
                .setAction("Action", null).show()
    }

}

     }

    func login() {
        Toast.makeText("正在登录，请稍后...").show()
        print("userName",self.userName)
        print("password",self.password)
        loginSuspend()
    }

    func loginSuspend() {
        DispatchQueue.global().async {
            print("Thread.mian",Thread.isMainThread,Thread.current)
        }
        DispatchQueue.main.asyncAfter(deadline: DispatchTime.now()+4, execute: {
        print("Thread.main",Thread.isMainThread,Thread.current)
        let loginResult: Void =  LoginService().loginAsync(userName:self.userName,userPassword:self.password, completionHandler:{(data,error) in
            if(error == nil){

               DispatchQueue.main.async {Toast.makeText( "出错了").show()}
            }
         })
        print("loginResult",loginResult)
        })

            if let result = result{

3.2 login.framework 分析

  • 我们在login module commonMain下的LoginService 中 分别声明了 3个函数方法：

fun login(userName: String?, userPassword: String?): LoginResult

该方法是一个普通方法，同步返回结果

suspend fun loginAsync(userName: String?, userPassword: String?): LoginResult

该方法是一个挂起函数，接受两个参数userName，userPassword

在android中配合协程使用可以直接通过 val loginJob = async { LoginService.loginAsync(email,password)}
val loginResult = loginJob.await() 来获取结果

在iOS中，suspend函数对应的会在函数参数的末尾位置，显示的声明一个completionHandler，类型为(LoginResult?, Error?) -> Void的callback 来接收回调结果

在iOS中启动suspend函数（启动协程）默认要在主线程中启动

fun doWork(taskId: Int, task : ()-> Unit, callback: ServiceCallBack?)

该方法是一个普通方法，最后一个参数为callback类型，通过callback异步接收回调结果
