一 如何在flutter中使用dart:io进行网络请求

dart:io 是系统提供的一套针对非Web应用程序，进行文件读取，套接字，HTTP和其他I/O支持的API。更多介绍详见dart:io库

dart:io 中提供了HttpServer和HttpClient两个核心类分别针对服务端开发和客户端请求开发，这些类提供了低级的HTTP功能，实际项目开发中官方提供了更为友好的类库,如：pub.dev上的http ,及flutterchina提供的 dio

1.1 使用HttpClient网络请求

  • step1 ： 创建HttpClient

    HttpClient httpClient = HttpClient();

  • step2 ： 打开HTTP连接

    HttpClientRequest request = await httpClient.getUrl(uri);

  • step3 ： 等待服务器连接和响应

    HttpClientResponse response = await request.close();

  • step4 ： 读取服务器响应内容

    int statusCode = response.statusCode;
if (statusCode == HttpStatus.ok) {
   String content =  await response.transform(const Utf8Decoder()).join();
}

  • step5 ： 断开连接

    httpClient.close();

1.2 HttpClient网络请求封装

  • 完整示例
    import 'dart:convert';
import 'dart:io';

class DartHttp {

   DartHttp._internal();
   
   static final DartHttp _instance = DartHttp._internal();
   // 工厂+静态变量 实现单例
   factory DartHttp() => _instance;

   Future<String>  get(String requestUrl, Map<String,dynamic> query) async {
     // 1.创建一个HttpClient
     HttpClient httpClient = HttpClient();
     // 设置连接超时时间
     httpClient.connectionTimeout = const Duration(seconds: 30);
     // 2. 设置请求地址，参数
     Uri org = Uri.parse(requestUrl);
     Uri uri = Uri(scheme: org.scheme,host: org.host,path:org.path,queryParameters: query);
     // 2. 打开Http连接
     HttpClientRequest request = await httpClient.getUrl(uri);
     // 3. 设置请求header
     request.headers.add("user-agent", "Chrome/104.0.0.0");
     // 4. 等待连接服务器
     HttpClientResponse response = await request.close();
     // 5. 读取响应内容
     String content = "";
     int statusCode = response.statusCode;
     if (statusCode == HttpStatus.ok) {
        String content =  await response.transform(const Utf8Decoder()).join();
     }else {
        content = "http status code: $statusCode";
     }
     // 6. 断开连接
     httpClient.close();
     return content;
   }
   
   Future<String>  post(String requestUrl, Map<String,dynamic> query) async {
      // 1.创建一个HttpClient
      HttpClient httpClient = HttpClient();
      // 设置连接超时时间
      httpClient.connectionTimeout = const Duration(seconds: 30);
      // 2. 设置请求地址，参数
      Uri org = Uri.parse(requestUrl);
      Uri uri = Uri(scheme: org.scheme,host: org.host,path:org.path,queryParameters: query);
      // 2. 打开Http连接
      HttpClientRequest request = await httpClient.postUrl(uri);
      // 3. 设置请求header
      request.headers.add("user-agent", "Chrome/104.0.0.0");
      // 4. 等待连接服务器
      HttpClientResponse response = await request.close();
      // 5. 读取响应内容
      String content = "";
      int statusCode = response.statusCode;
      if (statusCode == HttpStatus.ok) {
         String content =  await response.transform(const Utf8Decoder()).join();
      }else {
         content = "http status code: $statusCode";
      }
      // 6. 断开连接
      httpClient.close();
      return content;
   } 
}

二 dio网络库使用及封装

dio是一个强大的Dart Http请求库，支持Restful API、FormData、拦截器、请求取消、Cookie管理、文件上传/下载、超时、自定义适配器等，更多信息详见dio中文文档

2.1 dio简介

  • dio是一款强大的dart网络库，提供了丰富可扩展的插件，及网络请求拦截器，供开发者使用，如：网络请求日志打印，网络缓存，cookie管理，如果对okhttp熟悉和了解的开发者来说，你一定不会感到陌生。同时dio同时适配器的方式，底层默认封装了系统的HttpClient,
    方便开发者进行切换和定制底层网络库，灵活性比较强。

2.5 封装GET,POST代理方法的具体实现

  • dio 默认提供了get,post,fetch 快捷方法，底层都是通过request来实现，因此我们也通过调用request来实现
     Future<T?> _realRequest<T>(
 String url,
 String method,                // HTTP请求方法："GET", "POST"
 {
 Map<String,dynamic>? query,   // get请求参数
 Map<String,dynamic>? headers, // 请求header
 dynamic data,                 // post请求参数
 bool contentTypeJson= false,  // 是否设置Content-Type: application/json,默认 application/x-www-form-urlencoded
 ErrorHandler? errorHandler,   // 捕获异常处理的handler
 bool showLoading=false        // 是否显示loading
 }) async {
 try{
     if (showLoading) {
     showLoadingDialog();
     }
     // 请求头组装，如果本次请求单独设置了Options，则本次请求会覆盖公共的BaseOptions设置的内容
     Options options = Options(method: method,headers: headers) ;
     options.copyWith(sendTimeout:baseOptions.connectTimeout,receiveTimeout: baseOptions.connectTimeout);
     // 设置content-type
     options.contentType = contentTypeJson ? Headers.jsonContentType : Headers.formUrlEncodedContentType;
     if (contentTypeJson) {
        // dynamic data 转json
        data = _convertRequestData(data);
     }
     Uri? orgUri;
     try{
         orgUri = Uri.parse(url);
         Map<String,String> orgQuery = orgUri.queryParameters;
         query ??= <String,dynamic>{};
         query.addAll(orgQuery);
     }catch(e){
         var exception = ApiException.from(e);
         // 如果业务方未处理异常，则直接抛出异常
         if (errorHandler?.call(exception) != true && globalErrorHandler?.call(exception) != true) {
            throw exception;
         }
     }
     late Response response;
     if (orgUri != null) {
         response = await _dio.requestUri(orgUri,data:data, options: options);
     }else {
     if (baseOptions.baseUrl.isEmpty) {
        ApiException exception = ApiException(ApiException.defaultErrorCode, "未设置BaseUrl");
        // 如果业务方未处理异常，则直接抛出异常
        if (errorHandler?.call(exception) != true) {
           throw exception;
        }
     }
     // 使用dio发起请求
     response = await _dio.request(url,queryParameters:query,data:data, options: options);
     }
     return _parseHttpResponse(response);
 }catch(e){
     var exception = ApiException.from(e);
     // 如果业务方未处理异常，则直接抛出异常
     if (errorHandler?.call(exception) != true && globalErrorHandler?.call(exception) != true) {
        throw exception;
     }
 }finally{
     hideLoadingDialog();
 }
 return null;
}

 String _convertRequestData(dynamic data) {
   if (data != null) {
      data = jsonDecode(jsonEncode(data));
   }
   return data;
 }

    T? _parseHttpResponse<T>(Response response){
   if (response.statusCode == 200) {
        if (T.toString() == (RawData).toString()) {
        RawData rawData = RawData();
        rawData.value = response.data;
        return rawData as T;
      }
      ApiResponse<T> apiResponse = ApiResponse<T>.fromJson(response.data);
      if (apiResponse.code == 200) {
         return apiResponse.data;
      }else{

      throw exception;
   }
}

2.7 dio网络请求异常处理

  • 异常处理：网络请求异常，dio会抛出DioError,根据DioErrorType可以自行处理和二次包装，这里重新定义一个ApiException类

    class ApiException implements Exception {
      final String? message;
      final int? code;
      String? stackInfo;
      static const defaultErrorCode = -1;
      static const defaultMessage = "未知异常";

      ApiException(this.code, this.message);
 
      factory ApiException.fromDioError(DioError error){
         switch (error.type){
           case DioErrorType.cancel:
              return ApiException(defaultErrorCode, "请求取消");
           case DioErrorType.connectTimeout:
              return ApiException(defaultErrorCode, "连接超时");
           case DioErrorType.sendTimeout:
              return ApiException(defaultErrorCode, "请求超时");
           case DioErrorType.receiveTimeout:
              return ApiException(-1, "响应超时");
         case DioErrorType.response:
         try{
              ApiResponse apiResponse = ApiResponse.fromJson(error.response?.data);
              if (apiResponse.code != null) {
                 return ApiException(apiResponse.code, apiResponse.message);
              }
              int? errCode = error.response?.statusCode;
              switch (errCode) {
                 case 400:
                    return ApiException(errCode, "请求语法错误");
                 case 401:
                    return ApiException(errCode!, "没有权限");
                 case 403:
                    return ApiException(errCode!, "服务器拒绝执行");
                 case 404:
                    return ApiException(errCode!, "无法连接服务器");
                 case 405:
                    return ApiException(errCode!, "请求方法被禁止");
                 case 500:
                    return ApiException(errCode!, "服务器内部错误");
                 case 502:
                    return ApiException(errCode!, "无效的请求");
                 case 503:
                    return ApiException(errCode!, "服务器异常");
                 case 505:
                    return ApiException(errCode!, "不支持HTTP协议请求");
                 default:
                    return ApiException(errCode, error.response?.statusMessage ?? '未知错误');
              }
          } on Exception catch(e){
              return ApiException(defaultErrorCode, defaultMessage);
         }
         default:
            return ApiException(defaultErrorCode, error.message);
         }
      }
      
      factory ApiException.from(dynamic exception){
         if(exception is DioError){
            return ApiException.fromDioError(exception);
         }else if(exception is ApiException){
            return exception;
         }else {
            var apiException = ApiException(defaultErrorCode, defaultMessage);
            apiException.stackInfo = exception?.toString();
            return apiException;
         }
      }
}

三 在flutter中使用websocket

dart:io系统类库中的WebSocket类提供对 Web 套接字协议的支持。这允许客户端和服务器应用程序之间的全双工通信。
Web 套接字服务器使用普通的 HTTP 服务器来接受 Web 套接字连接。初始握手是一个 HTTP 请求，然后升级为 Web 套接字连接。

服务器使用WebSocketTransformer升级请求 并侦听返回的 Web 套接字上的数据。

客户端使用WebSocket.connect方法和使用 Web Socket 协议的 URI连接到WebSocket 。客户端可以使用WebSocket.add方法写入WebSocket。

pub.dev 提供了 web_socket_channel 库，对WebSocket进行了包装，提供了一套跨平台的WebSocketChannel的 API，本例将使用该库进行说明

3.1 WebSocket的通信5个步骤

  1. 连接到WebSocket服务器
  • 为了方便管理，我们自定义一个WebSocketManager，对WebSocket通信进行统一封装
    late IOWebSocketChannel _channel;

StreamController<WebSocketStatus>? socketStatusController = StreamController();
// socket 连接状态
WebSocketStatus connectStatus = WebSocketStatus.close;

Future connect(String url) async {
if (connectStatus == WebSocketStatus.close) {
   connectStatus = WebSocketStatus.connecting;
   socketStatusController ??= StreamController();
   socketStatusController?.add(WebSocketStatus.connecting);
   // 建立websocket连接
   _channel = IOWebSocketChannel.connect(Uri.parse(url));
   connectStatus = WebSocketStatus.connect;
   // 更新连接状态为已连接
   socketStatusController?.add(WebSocketStatus.connect);
   return true;
 }
}
  1. 监听来自服务器的消息
  • IOWebSocketChannel.stream.listen()监听数据，也可以结合StreamBuilder来实现异步更新ui
    StreamBuilder(
   stream: widget.channel.stream,
   builder: (context, snapshot) {
    // 通过hasData 接受数据
    return Text(snapshot.hasData ? '${snapshot.data}' : '');
   },
);
  1. 向服务端发送数据
  • 通过IOWebSocketChannel.sink.add(data),一个StreamSink类型的对象，用于接受流事件
    bool send(String content){
  if (connectStatus == WebSocketStatus.connect) {
    _channel.sink.add(content);
    return true;
  }
  return false;
}
  1. 关闭WebSocket连接
  • 通过IOWebSocketChannel.sink.close(),后续将不再添加任何流事件
    
Future disconnect() async {
if(connectStatus==WebSocketStatus.connect){
   connectStatus=WebSocketStatus.closing;
   socketStatusController?.add(WebSocketStatus.closing);
   await _channel.sink.close(3000,"主动关闭");
   connectStatus=WebSocketStatus.close;
   socketStatusController?.add(WebSocketStatus.close);
  }
}
