Android Q 文件存储兼容方案

Android 文件存储
Android
发布日期:   2021-09-26

背景

为了让用户更好地控制自己的文件，并限制文件混乱的情况，Android Q修改了APP访问外部存储中文件的方法。外部存储的新特性被称为Scoped Storage。

Android Q仍然使用READ_EXTERNAL_STORAGE和WRITE_EXTERNAL_STORAGE作为面向用户的存储相关运行时权限，但现在即使获取了这些权限，访问外部存储也受到了限制。

APP需要这些运行时权限的情景发生了变化，且各种情况下外部存储对APP的可见性也发生了变化。

在Scoped Storage新特性中，外部存储空间被分为两部分：

  • 公共目录：Downloads、Documents、Pictures 、DCIM、Movies、Music、Ringtones等
    公共目录下的文件在APP卸载后，不会删除。
    APP可以通过SAF(System Access Framework)、MediaStore接口访问其中的文件。
  • App-specific目录
    APP卸载后，数据会清除。
    APP的私密目录，APP访问自己的App-specific目录时无需任何权限
  • OPPO关于android Q文件储存变更描述
  • Google关于android Q文件存储变更描述

一、Android Q 文件存储行为变更

  • 无法直接在SD卡创建文件夹

  • 无法直接访问SD公有目录（Downloads、Documents、Pictures 、DCIM、Movies、Music、Ringtones等）的文件

  • API变更，Environment.getExternalStorageDirectory() 在android Q及以上版本不再适用，需要使用MediaStore接口

二、文件操作API示例：

  • 更多用法参考：com.yt.utils.MediaStoreUtils ,已经对MediaStore接口做了相应封装，另外，由于api限制，保存文件时需要对 Build.VERSION.SDK_INT进行判断（android 10 以上走MediaStore，10以下直接通过File IO操作
  • 如果不需要共享的文件，尽量使用context.getExternalCacheDir();获取公共缓存目录，需要申请sd卡存储权限）

2.1 查询文件URI

   public static Uri queryMediaStoreImageUri(Context context, MediaQuery query){
    if (context == null || query == null || !query.validRequest()){
        return null;
    }
    // 查询video文件 请使用 MediaStore.Video.Media.EXTERNAL_CONTENT_URI;
    Uri externalUri = MediaStore.Images.Media.EXTERNAL_CONTENT_URI;
    ContentResolver resolver = context.getContentResolver();
    Cursor cursor = resolver.query(externalUri, query.projection, query.selection,query.args,null);
    Uri imageUri = null;
    if (cursor != null && cursor.moveToFirst()){
        imageUri = ContentUris.withAppendedId(externalUri,cursor.getLong(0));
        cursor.close();
    }
    return imageUri;
}

2.2 读取文件

@WorkerThread
public static File copyMediaStoreToExternalCache(Context context,Uri imageUri) {
    File file = null;
    if (context == null || imageUri == null){
        return null;
    }
    String dateStamp = new SimpleDateFormat("yyyyMMdd_HHmmss", Locale.CHINA).format(new Date());
    long mills = System.currentTimeMillis();
    File dir = context.getExternalCacheDir();
    String mimeType = getMime(context,imageUri);
    String fileName = "hipac_cache_"+dateStamp+"_"+mills+"."+mimeType;
    if (!dir.exists()){
        dir.mkdirs();
    }
    try {
        file = File.createTempFile("hipac_cache_","."+mimeType,dir);
    } catch (IOException e) {
        file = new File(dir,fileName);
        if (!file.exists()){
            try {
                file.createNewFile();
            } catch (IOException ioException) {
                ioException.printStackTrace();
            }
        }
    }
    // 打开文件输入流
    try(InputStream input = context.getContentResolver().openInputStream(imageUri);
        FileOutputStream out = new FileOutputStream(file);
        BufferedOutputStream outBuffer = new BufferedOutputStream(out)) {
        int len =0;
        byte[] buffer = new byte[4096];
        while ((len=input.read(buffer)) !=-1){
            outBuffer.write(buffer,0,len);
        }
        outBuffer.flush();
    }catch (Exception e){
        Logs.e("ImageUtils","readMediaStoreImage() error="+e);
    }
    return file;
 }

2.3 将图片文件转成Bitmap


@WorkerThread
public static Bitmap readMediaStoreImage(Context context, Uri imageUri, BitmapFactory.Options options){
    Bitmap bitmap = null;
    if (context == null || imageUri == null){
        return null;
    }
    // "r" 表示读，"w"表示写，
    try(ParcelFileDescriptor pdf = context.getContentResolver().openFileDescriptor(imageUri,"r")) {
        bitmap = BitmapFactory.decodeFileDescriptor(pdf.getFileDescriptor(),null,options);
    }catch (Exception e){
        Logs.e("ImageUtils","readMediaStoreImage() error="+e);
    }
    return bitmap;
}

2.4 保存图片/视频到相册

@WorkerThread
  public static void insertImageToMediaStore(Context context, File sourceFile){
      if (context == null || sourceFile == null || !sourceFile.exists()){
          return;
      }
      String fileName = sourceFile.getName();
      String[] names = fileName.split("\\.");
      String suffix = names[names.length-1];
      Pair<String, String> mimePair = getMime(fileName);
      String mime = mimePair.first;
      boolean isVideo = "video".equals(mimePair.second);
      ContentValues contentValues = new ContentValues();
      contentValues.put(MediaStore.Images.Media.DISPLAY_NAME,sourceFile.getName());
      contentValues.put(MediaStore.Images.Media.TITLE,sourceFile.getName());
      contentValues.put(MediaStore.Images.Media.DATE_TAKEN,System.currentTimeMillis()+"");
      if (!mime.isEmpty()){
          contentValues.put(MediaStore.Images.Media.MIME_TYPE,mime);
      }
      if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q){
          // android Q 不再使用DATA字段，应该使用RELATIVE_PATH代替，相对于系统公有目录的路径,eg: "Pictures/hipac"
          String relativePath = isVideo ? "DCIM" : "Pictures/hipac";
          contentValues.put(MediaStore.Images.Media.RELATIVE_PATH,relativePath);
          Uri externalUri = isVideo ? MediaStore.Video.Media.EXTERNAL_CONTENT_URI : MediaStore.Images.Media.EXTERNAL_CONTENT_URI;
          Uri insertUri = context.getContentResolver().insert(externalUri,contentValues);
          if (insertUri != null){
              try (FileInputStream input = new FileInputStream(sourceFile);
                   OutputStream out = context.getContentResolver().openOutputStream(insertUri);
                   BufferedOutputStream bufferWriter = new BufferedOutputStream(out)){
                  int len=0;
                  byte[] buffer = new byte[4096];
                  while ((len=input.read(buffer)) !=-1){
                      bufferWriter.write(buffer,0,len);
                  }
                  bufferWriter.flush();
              }catch (Exception e){
                  Logs.e("ImageUtils","insertImageToMediaStore() error="+e);
              }
          }
      }else {
        // android 10 以下用insert方法，是直接插入到/storage/sdcard0，相册选择可能有问题
          contentValues.put(MediaStore.Images.Media.DATA,sourceFile.getAbsolutePath());
          File externalDir = isVideo ? Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_DCIM) : Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_PICTURES);
          File parent = new File(externalDir,"hipac");
          if (!parent.exists()){
              parent.mkdirs();
          }
          File destFile = new File(parent,sourceFile.getName());
          if (!destFile.exists()){
              try {
                  destFile.createNewFile();
              } catch (IOException e) {
                  Logs.e("ImageUtils","insertImageToMediaStore() error="+e);
              }
          }
          try(FileInputStream input = new FileInputStream(sourceFile);
              FileOutputStream out = new FileOutputStream(destFile);
              BufferedOutputStream outBuffer = new BufferedOutputStream(out)) {
              int len =0;
              byte[] buffer = new byte[4096];
              while ((len=input.read(buffer)) !=-1){
                  outBuffer.write(buffer,0,len);
              }
              outBuffer.flush();
              context.sendBroadcast(new Intent(Intent.ACTION_MEDIA_SCANNER_SCAN_FILE,Uri.fromFile(destFile)));
          } catch (Exception e) {
              Logs.e("ImageUtils","insertImageToMediaStore() error="+e);
          }
      }

  }
